    \begin{aligned} 
        f_1(n) &= (n - h(n)) \ShiftLeft 1 + 1\\% Todd
        %
        f_2(n) &= \BitNeg h(2n) \BitAnd ((n \ShiftLeft 1) \ShiftRight 1)\\% Wikipedia
        %
        f_3(n) &= (n \BitAnd \BitNeg (1 \ShiftLeft g(n))) \ShiftLeft 1 + 1\\% Dillon
		\\
		\text{Where}\\%
        g(n) &= \lfloor\log_2(n)\rfloor\\%
        h(n) &= n \BitAnd (1 \ShiftLeft g(n))%
    \end{aligned}
